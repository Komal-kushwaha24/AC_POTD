class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {
      int n=digits.size();
      for(int i=n-1;i>=0;i--){
        int dig=digits[i];
        if(dig<9){
            digits[i]++;
            return digits;
        }
        digits[i]=0;
      }  
      digits.insert(digits.begin(),1);
      return digits;
    }
};
